CREATE OR REPLACE PROCEDURE P_AUTHENTICATION 
(
  USERNAME IN VARCHAR2 , PASSWORD IN VARCHAR2, RESULT OUT VARCHAR2) AS
  CNT NUMBER(2);
BEGIN
   SELECT COUNT(*) INTO CNT FROM userinfo WHERE UNAME=USERNAME AND PWD=PASSWORD;
   IF(CNT<>0) THEN
   RESULT:='VALID CREDENTIAL';
   ELSE
   RESULT:='INVALID CREDENTIAL';
   END IF;
END P_AUTHENTICATION;
